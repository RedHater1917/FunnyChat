CREATE ROLE weaponsystem WITH
  LOGIN
  SUPERUSER
  INHERIT
  CREATEDB
  CREATEROLE
  REPLICATION
  ENCRYPTED PASSWORD 'md5aa83fcb314073c8a8b9702946dae7d68';
  
CREATE DATABASE weaponsystem
    WITH 
    OWNER = weaponsystem
    ENCODING = 'UTF8'
    LC_COLLATE = 'Russian_Russia.1251'
    LC_CTYPE = 'Russian_Russia.1251'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1;

CREATE SCHEMA IF NOT EXISTS weaponsystem;
create table if not exists weaponsystem.images(
    Id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name varchar(255),
    pic bytea,
    type varchar(255)
);
create table if not exists weaponsystem.book(
    Id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    NAME varchar(255),
    HANDS_COUNT integer,
    AVATARID bigint REFERENCES weaponsystem.images(id),
    DESCRIPTION varchar(255)
);
create table if not exists weaponsystem.weaponrating(
	Id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
	WeaponId bigint REFERENCES weaponsystem.book(id),
	REQUIREMENTS_FOR_STRENGTH_AND_ENDURANCE double precision,
	REQUIREMENT_FOR_EXP double precision,
	RANGE_ double precision,
	DURABILITY double precision,
	REPAIR_ABILITY double precision,
	PACE double precision,
	SINGLE_HIT_SPEED double precision,
	DEFENSE double precision,
	ACCURACY double precision,
	FINT_VARIETY double precision,
	SAFETY double precision
)
